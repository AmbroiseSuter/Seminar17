\def\scalefactor{1.04}
\def\gridspacing{0.5}
\def\xorigin{0}
\def\yorigin{0}

\newcommand\punkt[2]{
\def\x{#1}
\def\y{#2}
\draw ({(\x + \xorigin) * \gridspacing},{(\y + \yorigin) * \gridspacing}) circle[radius=1mm]{};
\draw[fill=black] ({(\x + \xorigin) * \scalefactor * \gridspacing}, {(\y + \yorigin) * \scalefactor * \gridspacing}) circle[radius=1mm]{};
}
